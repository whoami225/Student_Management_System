    <script>
        function loadStudents() {
            var subjectId = document.getElementById("subject").value;
            if(subjectId) {
                window.location.href = "attendance?action=mark&subject=" + subjectId;
            }
        }
    </script>

    <h2>Mark Attendance</h2>
    
    <c:if test="${not empty message}">

    </c:if>
    
    <div class="form-group">
        <label for="subject">Select Subject:</label>
        <select id="subject" name="subject" onchange="loadStudents()">
            <option value="">-- Select Subject --</option>
            <c:forEach items="${subjects}" var="subject">
                <option value="${subject.subjectId}" 
                    <c:if test="${param.subject == subject.subjectId}">selected</c:if>>
                    ${subject.subjectName}
                </option>
            </c:forEach>
        </select>
    </div>
    
    <c:if test="${not empty students}">
        <form action="attendance" method="post">
            <input type="hidden" name="action" value="mark">
            <input type="hidden" name="subject" value="${param.subject}">
            
            <table>
                <thead>
                    <tr>
                        <th>Student ID</th>
                        <th>Student Name</th>
                        <th>Status</th>
                    </tr>
                </thead>
                <tbody>
                    <c:forEach items="${students}" var="student">
                        <tr>
                            <td>${student.userId}
                                <input type="hidden" name="studentId" value="${student.userId}">
                            </td>
                            <td>${student.firstName} ${student.lastName}</td>
                            <td>
